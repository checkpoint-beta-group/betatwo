10  	private final int height;
11  	private final int width;
12  	private final Tile[][] tiles;
13  	
14  	/**
15  	 * Create a new board.
16  	 * @param w Width of the board
17  	 * @param h Height of the board.
18  	 */
19  	public Board(int w, int h) {
20  		assert w >= 0;
21  		assert h >= 0;
22  		width = w;
23  		height = h;
24  		tiles = new Tile[width][height];
25  		
26  		for (int x = 0; x < w; x++) {
27  			for (int y = 0; y < h; y++) {
28  				tiles[x][y] = new Tile(x, y); 
29  			}
30  		}
31  	}
32  	
33  	@Override
34  	public int getHeight() {
35  		return height;
36  	}
37  
38  	@Override
39  	public int getWidth() {
40  		return width;
41  	}

43  	/**
44  	 * Put a sprite at a given position.
45  	 * @param s Sprite to be put on the board
46  	 * @param x x-coordinate
47  	 * @param y y-coordinate
48  	 */
49  	public void put(Sprite s, int x, int y) {
50  		assert withinBorders(x, y) : "PRE1: (x,y) on board.";
51  		assert s != null : "PRE2: Sprite not null";
52  		s.occupy(tileAt(x, y));
53  	}
54  	
55  	/**
56  	 * Verify that the given location falls within the
57  	 * borders of the board.
58  	 * @param x x-coordinate
59  	 * @param y y-coordinate
60  	 * @return True iff (x,y) falls within the board.
61  	 */
62  	public boolean withinBorders(int x, int y) {
63  		return
64  			x >= 0 && x < width 
65  			&& y >= 0 && y < height;
66  	}
67  
68  	@Override
69  	public Sprite spriteAt(int x, int y) {
70  		assert withinBorders(x, y) : "PRE: (x,y) on board.";		
71  		return tileAt(x, y).topSprite();
72  	}
73  	
74  	@Override
75  	public SpriteType spriteTypeAt(int x, int y) {
76  		Sprite s = spriteAt(x, y);
77  		SpriteType result;
78  		if (s == null) {
79  			result = SpriteType.EMPTY;
80  		} else {
81  			result = s.getSpriteType();
82  		}
83  		return result;
84  	}

86  	/**
87  	 * @param x x-coordinate
88  	 * @param y y-coordinate
89  	 * @return The tile at the given (x,y) place
90  	 */
91  	public Tile tileAt(int x, int y) {
92  		return tiles[x][y];
93  	}
94  	
95  	/**
96  	 * Return a tile at position (x+dx, y+dy) from current
97  	 * tile at (x,y).
98  	 * @param t Reference tile / starting point.
99  	 * @param dx delta in x direction
100 	 * @param dy delta in y direction
101 	 * @return Tile at (x+dx, y+dy)
102 	 */
103 	public Tile tileAtOffset(Tile t, int dx, int dy) {
104 		assert t != null;
105         int newx = tunnelledCoordinate(t.getX(), getWidth(), dx);
106         int newy = tunnelledCoordinate(t.getY(), getHeight(), dy);
107         assert withinBorders(newx, newy);
108         Tile result = tileAt(newx, newy);
109         assert result != null;
110         return result;
111 	}
112 	
113 	private int tunnelledCoordinate(int current, int max, int delta) {
114 		assert current >= 0;
115 		assert current < max;
116 
117 		// additional max needed if (current + delta) < 0.
118 		int result = ((current + delta) % max + max) % max;
119 
120 		assert result >= 0;
121 		assert result < max;
122 
123 		return result;
124 	}
125 	
126 	/**
127 	 * Obtain the tile in the given direction.
128 	 * @param t Starting position
129 	 * @param dir Direction
130 	 * @return Tile in direction from the given starting position.
131 	 */
132 	public Tile tileAtDirection(Tile t, Direction dir) {
133 		return tileAtOffset(t, dir.getDx(), dir.getDy());
134 	}
